function trackOthers() { const bl = document.getElementById('blNumber').value.trim(); if (!bl) { alert('Please enter a B/L or container number first.'); document.getElementById('blNumber').focus(); return; } fireConfetti(); const commonCarriers = [ "ZIM - www.zim.com", "Yang Ming - www.yangming.com", "Wan Hai - www.wanhai.com", "PIL - www.pilship.com" ]; const userUrl = prompt( "Tracking for other carriers:\n\n" + "Common carriers:\n" + commonCarriers.join("\n") + "\n\n" + "Option A: Paste the tracking URL\n" + "Option B: Leave blank to search on Google\n\n" + "Tracking: " + bl, "" ); if (userUrl === null) return; // 用户点击了取消 if (userUrl.trim() === "") { // 方案 B: 如果输入为空，直接去 Google 搜索该提单号 window.open(`https://www.google.com/search?q=${encodeURIComponent(bl)}+container+tracking`, '_blank'); } else { // 方案 A: 如果用户输入了网址，自动加上 https:// 前缀（如果用户没写的话）并打开 let finalUrl = userUrl.trim(); if (!/^https?:\/\//i.test(finalUrl)) { finalUrl = 'https://' + finalUrl; } window.open(finalUrl, '_blank'); } }
